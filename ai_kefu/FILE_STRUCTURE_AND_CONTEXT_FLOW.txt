Current pattern in executor.py [Lines 596-641]:

def _get_summary_cache_key(self, chat_id: str) -> str:
    return f"history_summary:{chat_id}"

def _get_cached_summary(self, chat_id: str) -> Optional[dict]:
    try:
        cache_key = self._get_summary_cache_key(chat_id)
        data = self.session_store.client.get(cache_key)  # Redis client
        if data:
            return json.loads(data)
    except Exception as e:
        logger.debug(f"Failed to read summary cache: {e}")
    return None

def _set_cached_summary(self, chat_id: str, summary: str, is_returning_customer: bool, fingerprint: dict):
    try:
        cache_key = self._get_summary_cache_key(chat_id)
        cache_data = json.dumps({
            "summary": summary,
            "is_returning_customer": is_returning_customer,
            "fingerprint": fingerprint
        }, ensure_ascii=False)
        # TTL 1 hour
        self.session_store.client.setex(cache_key, 3600, cache_data)
    except Exception as e:
        logger.debug(f"Failed to cache summary: {e}")

For item info, apply the same pattern:
  - Cache key: f"item_info:{item_id}"
  - TTL: 3600 (1 hour) or higher depending on data freshness needs
  - Store JSON in Redis
  - Fingerprint: message_count + last_message_at for invalidation
